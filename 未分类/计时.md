/**
 * 计算某农历年的总天数（含闰月）
 * @param y 农历年（如2026）
 * @returns 该年总天数（353-385天之间）
 */
Calendar.prototype.lYearDays = function(y) {
    var i, sum = 348;  // 基础12个月，每月29天，共348天
    // 遍历12个月，检查每个月是29天还是30天
    for(i=0x8000; i>0x8; i>>=1) {
        sum += (this.lunarInfo[y-1900] & i) ? 1 : 0;
    }
    // 加上闰月天数
    return(sum + this.leapDays(y));
};

/**
 * 计算某农历年闰月的天数
 * @param y 农历年
 * @returns 闰月天数（29或30），无闰月返回0
 */
Calendar.prototype.leapDays = function(y) {
    if(this.leapMonth(y)) {
        // 检查闰月天数位（第16位）
        return((this.lunarInfo[y-1900] & 0x10000) ? 30 : 29);
    }
    else return(0);
};

/**
 * 获取某农历年的闰月月份
 * @param y 农历年
 * @returns 闰月月份（1-12），无闰月返回0
 */
Calendar.prototype.leapMonth = function(y) {
    // 取低4位
    return(this.lunarInfo[y-1900] & 0xf);
};

/**
 * 计算某农历年某月的天数
 * @param y 农历年
 * @param m 农历月（1-12）
 * @returns 该月天数（29或30）
 */
Calendar.prototype.monthDays = function(y,m) {
    // 检查第m位是否为1（30天）
    return((this.lunarInfo[y-1900] & (0x10000>>m)) ? 30 : 29);
};

/**
 * 将偏移量转换为干支
 * @param offset 天干地支的循环偏移量
 * @returns 干支字符串，如"甲子"
 */
Calendar.prototype.toGanZhi = function(offset) {
    return this.Gan[offset % 10] + this.Zhi[offset % 12];
};

/**

    var offDate = new Date((31556925974.7*(y-1900)+this.sTermInfo[n]*60000)+Date.UTC(1900,0,6,2,5));
    return(offDate.getUTCDate());
};

/**
 * 核心函数：公历转农历
 * @param y 公历年
 * @param m 公历月（1-12）
 * @param d 公历日（1-31）
 * @returns 包含农历信息的对象
 */
Calendar.prototype.solar2lunar = function(y, m, d) {
    // 只支持1900-2100年
    if(y<1900 || y>2100) return -1;

    // 创建目标日期对象
    var objDate = new Date(y, parseInt(m)-1, parseInt(d));

    var i, leap=0, temp=0;
    // 基准日期：1900年1月31日（农历1900年正月初一）
    var baseDate = new Date(1900,0,31);

    // 计算目标日期与基准日期之间的天数差
    var offset = (objDate - baseDate) / 86400000;

    // 逐年减去天数，确定农历年
    for(i=1900; i<2101 && offset>0; i++) { 
        temp = this.lYearDays(i);   // 该农历年总天数
        offset -= temp; 
    }

    // 如果减多了，回退一年
    if(offset < 0) { 
        offset += temp; 
        i--; 
    }

    var year = i;  // 农历年

    // 获取该年的闰月
    var leap = this.leapMonth(i);
    var isLeap = false;  // 当前是否在闰月

    // 逐月减去天数，确定农历月
    for(i=1; i<13 && offset>0; i++) {
        // 如果该年有闰月，且当前月份是闰月后一个月，先处理闰月
        if(leap>0 && i==(leap+1) && isLeap==false) { 
            --i; 
            isLeap = true; 
            temp = this.leapDays(year);  // 闰月天数
        } else { 
            temp = this.monthDays(year, i);  // 普通月天数
        }

        // 闰月只处理一次
        if(isLeap==true && i==(leap+1)) isLeap = false;
        offset -= temp;
    }

    // 边界处理：如果offset正好为0且下一个月是闰月
    if(offset==0 && leap>0 && i==leap+1) {
        if(isLeap) { 
            isLeap = false; 
        } else { 
            isLeap = true; 
            --i; 
        }
    }

    // 如果减多了，回退一个月
    if(offset<0){ 
        offset += temp; 
        --i; 
    }

    var month = i;  // 农历月
    // 修复：day必须是整数（offset可能是浮点数）
    var day = Math.floor(offset + 1);  // 农历日

    // ========== 计算干支 ==========

    // 年干支：基于1900年（庚子年）的偏移
    var gzY = this.toGanZhi(y - 1900 + 36);

    // 月干支
    var firstNode = this.getTerm(y, m*2);  // 当月节气（中气）
    var gzM = this.toGanZhi((y-1900)*12 + m + 12);
    // 如果已经过了节气，月干支进一位
    if(d >= firstNode) gzM = this.toGanZhi((y-1900)*12 + m + 13);

    // 日干支：基于1900年1月31日的偏移
    var dayCyclical = Date.UTC(y, m-1, 1, 0, 0, 0, 0) / 86400000 + 25567 + 10;
    var gzD = this.toGanZhi(dayCyclical + d - 1);

    // 生肖：基于年支
    var animal = this.Animals[(y - 4) % 12];

    // 节气判断
    var isTerm = false;
    var Term = null;
    var term3 = this.getTerm(y, (m-1)*2);    // 当月第一个节气
    var term2 = this.getTerm(y, (m-1)*2+1);  // 当月第二个节气
    if(d == term3) { isTerm = true; Term = this.solarTerm[(m-1)*2]; }
    if(d == term2) { isTerm = true; Term = this.solarTerm[(m-1)*2+1]; }

    // 组装农历日期字符串
    var lunarDate = this.nStr2[month-1] + "月" + this.toChinaDay(day);

    // 返回结果对象
    return {
        lYear: year,           // 农历年
        lMonth: month,         // 农历月
        lDay: day,             // 农历日
        animal: animal,        // 生肖
        IMonthCn: (isLeap?"闰":"") + this.nStr2[month-1]+"月",  // 农历月中文
        IDayCn: this.toChinaDay(day),  // 农历日中文
        cYear: gzY,            // 年干支
        cMonth: gzM,           // 月干支
        cDay: gzD,             // 日干支
        isLeap: isLeap,        // 是否闰月
        ncWeek: "星期"+"日一二三四五六".charAt(objDate.getDay()),  // 星期
        isTerm: isTerm,        // 当天是否有节气
        Term: Term,            // 节气名称
        lunarDate: lunarDate   // 完整农历日期
    };
};

/**
 * 根据当前时间和日干计算时辰干支
 * @param hour 当前小时（0-23）
 * @param minute 当前分钟
 * @param dayGanZhi 当日干支（如"丙辰"）
 * @returns 时辰信息对象
 */
function getShiChen(hour, minute, dayGanZhi) {
    // 确定当前在哪个时辰（每个时辰2小时）
    var zhiIndex;
    if (hour >= 23 || hour < 1) zhiIndex = 0;      // 子时：23:00-01:00
    else if (hour >= 1 && hour < 3) zhiIndex = 1;  // 丑时
    else if (hour >= 3 && hour < 5) zhiIndex = 2;  // 寅时
    else if (hour >= 5 && hour < 7) zhiIndex = 3;  // 卯时
    else if (hour >= 7 && hour < 9) zhiIndex = 4;  // 辰时
    else if (hour >= 9 && hour < 11) zhiIndex = 5; // 巳时
    else if (hour >= 11 && hour < 13) zhiIndex = 6; // 午时
    else if (hour >= 13 && hour < 15) zhiIndex = 7; // 未时
    else if (hour >= 15 && hour < 17) zhiIndex = 8; // 申时
    else if (hour >= 17 && hour < 19) zhiIndex = 9; // 酉时
    else if (hour >= 19 && hour < 21) zhiIndex = 10; // 戌时
    else zhiIndex = 11; // 亥时：21:00-23:00

    var zhi = calendar.Zhi[zhiIndex];  // 地支

    // 时辰天干根据日干推算：
    // 甲己日起甲子时，乙庚日起丙子时，丙辛日起戊子时，丁壬日起庚子时，戊癸日起壬子时
    var dayGan = dayGanZhi.charAt(0);  // 日干
    var startGanIndex;  // 该日干对应的子时天干索引
    switch(dayGan) {
        case '甲': case '己': startGanIndex = 0; break; // 甲
        case '乙': case '庚': startGanIndex = 2; break; // 丙
        case '丙': case '辛': startGanIndex = 4; break; // 戊
        case '丁': case '壬': startGanIndex = 6; break; // 庚
        case '戊': case '癸': startGanIndex = 8; break; // 壬
        default: startGanIndex = 0;
    }

    // 根据时辰偏移计算天干
    var ganIndex = (startGanIndex + zhiIndex) % 10;
    var gan = calendar.Gan[ganIndex];

    var shichenNames = ["子时", "丑时", "寅时", "卯时", "辰时", "巳时", "午时", "未时", "申时", "酉时", "戌时", "亥时"];

    return {
        name: shichenNames[zhiIndex],      // 时辰名称
        ganZhi: gan + zhi,                // 时辰干支
        zhiIndex: zhiIndex,               // 地支索引
        timeRange: getShiChenTimeRange(zhiIndex)  // 时间范围
    };
}

/**
 * 获取时辰的时间范围字符串
 */
function getShiChenTimeRange(zhiIndex) {
    var ranges = [
        "23:00-01:00", "01:00-03:00", "03:00-05:00", "05:00-07:00",
        "07:00-09:00", "09:00-11:00", "11:00-13:00", "13:00-15:00",
        "15:00-17:00", "17:00-19:00", "19:00-21:00", "21:00-23:00"
    ];
    return ranges[zhiIndex];
}

// ========== 古代计时法 ==========

/**
 * 计算古代计时表示（子初/子正 + 一刻/二刻等）
 * 
 * 古代计时规则：
 * - 一天分为12个时辰，每个时辰2小时
 * - 每个时辰分为"初"和"正"，各1小时
 * - 每1小时分为4刻，每刻约15分钟
 * - 如：寅正三刻 = 寅时（3-5点）的后一小时（4-5点）的第3刻（4:30-4:45）
 */
function getGuJiTime(hour, minute, second) {
    // 计算从子时初（23:00）开始的总分钟数
    var totalMinutesOfDay;
    if (hour >= 23) {
        totalMinutesOfDay = (hour - 23) * 60 + minute;
    } else {
        totalMinutesOfDay = (hour + 1) * 60 + minute;
    }

    // 确定当前时辰索引（0-11）
    var zhiIndex = Math.floor(totalMinutesOfDay / 120) % 12;
    var zhiNames = ["子", "丑", "寅", "卯", "辰", "巳", "午", "未", "申", "酉", "戌", "亥"];
    var zhi = zhiNames[zhiIndex];

    // 时辰内的分钟数（0-119）
    var minutesInShichen = totalMinutesOfDay % 120;

    // 初/正：前60分钟为"初"，后60分钟为"正"
    var chuZheng = minutesInShichen < 60 ? "初" : "正";

    // 计算刻数：每15分钟为1刻
    var keInHalf = Math.floor((minutesInShichen % 60) / 15) + 1;
    if (keInHalf > 4) keInHalf = 4;
    var keNames = ["", "一刻", "二刻", "三刻", "四刻"];
    var ke = keNames[keInHalf];

    return {
        shichenZheng: zhi + chuZheng,   // 如"寅正"
        ke: ke,                          // 如"三刻"
        detail: zhi + chuZheng + ke,    // 如"寅正三刻"
        zhiIndex: zhiIndex
    };
}

// ========== 时钟更新主函数 ==========

/**
 * 更新页面所有显示内容
 * 每秒执行一次
 */
function updateClock() {
    var now = new Date();

    // 数字补零函数：8 -> "08"
    var pad = function(n) { return String(n).padStart(2, '0'); };

    // 获取当前时间各部分
    var y = now.getFullYear();      // 年
    var m = now.getMonth() + 1;     // 月（0-11，+1变成1-12）
    var d = now.getDate();          // 日
    var h = now.getHours();         // 时
    var min = now.getMinutes();     // 分
    var sec = now.getSeconds();     // 秒
    var w = now.getDay();           // 星期（0=周日）

    // ========== 更新公历显示 ==========
    document.getElementById('solarDate').textContent = 
        y + '年' + pad(m) + '月' + pad(d) + '日';
    document.getElementById('weekday').textContent = 
        '星期' + '日一二三四五六'.charAt(w);
    document.getElementById('clock').textContent = 
        pad(h) + ':' + pad(min) + ':' + pad(sec);

    // ========== 计算并更新农历显示 ==========
    var lunar = calendar.solar2lunar(y, m, d);

    // 农历日期：如"农历 四月廿六"
    document.getElementById('lunarDate').textContent = 
        '农历 ' + lunar.IMonthCn + lunar.IDayCn;

    // 干支：如"丙午年 癸未月 丙辰日"
    document.getElementById('ganzhi').textContent = 
        lunar.cYear + '年 ' + lunar.cMonth + '月 ' + lunar.cDay + '日';

    // 生肖
    document.getElementById('animal').textContent = 
        '生肖：' + lunar.animal;

    // ========== 时辰干支 ==========
    var sc = getShiChen(h, min, lunar.cDay);
    document.getElementById('shichen').textContent = 
        '时辰：' + sc.ganZhi + '（' + sc.name + ' ' + sc.timeRange + '）';

    // ========== 古代计时法 ==========
    var gj = getGuJiTime(h, min, sec);
    document.getElementById('guji').textContent = '古制：' + gj.detail;

    // ========== 节气（当天有则显示）==========
    document.getElementById('term').textContent = 
        lunar.isTerm ? '今日节气：' + lunar.Term : '';
}

// 页面加载时立即执行一次
updateClock();

// 每秒自动更新（1000毫秒 = 1秒）
setInterval(updateClock, 1000);
